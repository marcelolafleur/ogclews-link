% one channel wire: #1=y #2=OGtext #3=chan-id #4=CLEWStext #5=edge style #6=pill border
\newcommand{\chanrow}[6]{%
  \node[ogobj] (L) at (0,#1) {#2};
  \node[clobj] (R) at (10.7,#1) {#4};
  \draw[#5] (L.east) -- (R.west);
  \node[chanpill, draw=#6, text=ink] at ($(L.east)!0.5!(R.west)$) {\texttt{#3}};
}
