\section{The coupling problem}

\begin{frame}{Two models, complementary blind spots}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\color{clews}\bfseries CLEWS / OSeMOSYS}\par\smallskip
      {\small a least-cost optimiser: chooses what to build and run so that energy
       demand is met at the lowest discounted system cost}\par\smallskip
      {\small\begin{itemize}
        \item \textbf{rich:} technology, season-by-season operation, emissions, land \& water
        \item \textbf{blind:} demand is \emph{fixed} --- no price response, no households,
              no government budget, no distribution
      \end{itemize}}
    \end{column}
    \begin{column}{0.48\textwidth}
      {\color{ogc}\bfseries OG-Core}\par\smallskip
      {\small overlapping-generations general equilibrium: households by age $S$
       $\times$ lifetime income $J$, $M$ industries, a government that closes its
       budget each period}\par\smallskip
      {\small\begin{itemize}
        \item \textbf{rich:} behaviour, saving, fiscal policy, incidence, demography
        \item \textbf{blind:} no energy technology --- energy is not a priced input
              to production
      \end{itemize}}
    \end{column}
  \end{columns}
  \medskip
  \centering{\color{sub}\small Each is detailed exactly where the other is blank ---
  so what usefully crosses the seam differs by direction:\\
  \textbf{quantities} flow from the economy, \textbf{prices and physical outcomes}
  flow from the energy system.}
\end{frame}

\dgmframe{The structural seam}{architecture}{%
  Quantities flow right (energy demand, the interest rate); prices and physical
  outcomes flow left (the energy price, build cost, emissions). The models stay
  separate and pass a few auditable signals.}

\begin{frame}{Why a \emph{staged soft} link, not one big model}
  \textbf{A hard link is ruled out structurally, not practically:}
  {\small\begin{itemize}
    \item \OGCore{} steps through time, period by period; the energy model plans
          all years at once
    \item neither fits inside the other --- \OGCore{} has no single objective the
          energy model could be nested in --- and no shock can be injected mid-solve
  \end{itemize}}
  \medskip
  \textbf{So the models stay separate and the link runs in stages:}
  \smallskip
  \begin{center}\small
  \resizebox{0.98\linewidth}{!}{%
  \begin{tikzpicture}[node distance=4mm]
    \node[wdg] (a) {solve \OGCore{}\\baseline};
    \node[sig, right=of a] (b) {solve \CLEWS{},\\extract signals};
    \node[wdg, right=of b] (c) {apply them to the\\economy; solve reform};
    \node[sig, right=of c] (d) {send results\\back to \CLEWS{}};
    \node[eff, right=of d] (e) {iterate to a\\fixed point};
    \draw[ch] (a) -- (b); \draw[ch] (b) -- (c); \draw[ch] (c) -- (d);
    \draw[ch, dashed] (d) -- (e);
  \end{tikzpicture}}
  \end{center}
  \smallskip
  {\color{sub}\small Validated today: the single pass (solid). The loop (dashed) is
  designed --- the return signals already exist --- but not yet wired.}
\end{frame}

\section{Signals, wedges, and the price that carries the load}

\begin{frame}{The vocabulary: signal $\to$ wedge $\to$ channel}
  \begin{description}[<+->]
    \item[{\color{clews}Signal}] a quantity read out of a \emph{solved} energy
      scenario: a fuel's marginal cost, the capital a build-out needs, emissions,
      an implied carbon price.
    \item[{\color{ogc}Wedge}] the \OGCore{} parameter that carries it: the
      consumption tax $\tauc$, an industry's productivity $Z_m$, public investment
      $\alphaI$, mortality $\rho(s,t)$, effective labour $e(t,s,j)$. (A ``wedge'' in
      the tax sense --- the simplest channel inserts one between producer and
      consumer price.)
    \item[Channel] one signal mapped to one wedge, with a guard that fires when the
      mapping is large or ill-posed.
  \end{description}
  \medskip
  \onslide<3->{%
  {\color{sub}\small Each channel is one small, testable mapping --- and nothing
  country-specific is built in: which industry is electricity, and how many
  industries there are, is read from the country's own model at run time.}}
\end{frame}

\begin{frame}{The signal that links the two models: the energy price}
  \footnotesize Of everything the energy model produces, \textbf{one number} does the
  real work across the seam --- the \textbf{price of energy}. (The model sets it by
  balancing supply and demand for every fuel in every season; what one more unit would
  cost to supply \emph{is} that price.) It is the right hand-off because it does two
  things the energy model, on its own, cannot:
  \smallskip
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\small\bfseries 1. It lets demand respond}\par
      {\footnotesize Inside the energy model, demand is a \emph{fixed} input. Hand the
      price to \OGCore{} households --- whose energy use responds to price, down to a
      subsistence floor --- and demand finally reacts.}
    \end{column}
    \begin{column}{0.48\textwidth}
      {\small\bfseries 2. It shows who pays}\par
      {\footnotesize A single ``system cost'' is silent on who bears it. \OGCore{}'s
      households, by age and income, split the burden --- and because energy is a
      partial necessity, a price rise falls hardest on the poor.}
    \end{column}
  \end{columns}
  \smallskip
  {\raggedright\color{sub}\footnotesize So the energy price is the hand-off: the economy
  reacts to it, and reveals who it hits. (The two models fully \emph{agree} once that
  price also equals the cost of supplying what the economy then demands --- what closing
  the loop, later, drives toward.)\par}
\end{frame}

\begin{frame}{The energy channel: electricity's two uses}
  \footnotesize We have the signal --- the energy price. But there is a catch: in
  \OGCore{}, firms produce with only \textbf{capital and labour; energy is not a line in
  their costs}. So the channel represents an electricity-price change by the \emph{two
  ways electricity is used}, and applies \textbf{both} as one transmission:
  \smallskip
  {\footnotesize
  \begin{itemize}\setlength\itemsep{4pt}
    \item[\textbf{1}] \textbf{as an input to firms} --- the bulk of electricity
      ($\sim$3/4 in the Philippine accounts): a cost-push, raising each industry's cost
      by the electricity it uses (directly and through what it buys). The contractionary,
      wage-bearing side --- it sets the output sign.
    \item[\textbf{2}] \textbf{as final consumption by households} --- the rest: a recycled
      consumption wedge on the energy good. The regressive cost-of-living side.
  \end{itemize}}
  \smallskip
  Electricity's own self-use is netted out, so the two sides do not double-count. Neither
  alone is the channel: the household wedge is $\approx$ output-neutral, the cost-push is
  contractionary --- only together do they carry the price into both the cost of living
  \emph{and} the cost of production.
  \smallskip
  {\raggedright\color{claret}\footnotesize The input side is a reduced-form proxy
  (\OGCore{} has no energy in production); the fully-structural version --- energy as a
  real priced input --- is the destination, not yet built.\par}
\end{frame}

\section{The eight channels}

\begin{frame}{The map}
  \centering\scriptsize
  \renewcommand{\arraystretch}{1.02}
  \begin{tabularx}{\textwidth}{@{}l c >{\raggedright\arraybackslash}X >{\raggedright\arraybackslash}X >{\raggedright\arraybackslash}X@{}}
  \toprule
  \textbf{Channel} & \textbf{Dir.} & \textbf{\CLEWS{} signal} & \textbf{\OGCore{} object} & \textbf{First-order effect} \\
  \midrule
  Energy price & {\color{clews}c$\to$o} & electricity price (marginal cost / index) & cost-push on firms + consumption wedge & output $\downarrow$; regressive incidence \\
  Carbon price & {\color{policy}pol} & carbon price (\$/tCO$_2$) & energy-wedge add-on + emissions penalty & demand $\downarrow$; emissions $\downarrow$ \\
  Public investment & {\color{clews}c$\to$o} & grid \& transmission build cost & public-investment share $\to$ public capital & public capital $\uparrow$; crowding-out, debt \\
  Capital intensity & {\color{clews}c$\to$o} & power-fleet capital-cost share & energy industry's capital share ($\gammam$) & energy price $\downarrow$; energy $K$ $\downarrow$ \\
  Investment incentive & {\color{policy}pol} & tax-credit rate (policy response) & cost of capital (investment tax credit) & capital reallocates into energy \\
  Health & {\color{clews}c$\to$o} & PM$_{2.5}$ emissions change & survival rates + worker productivity & mortality $\approx 0$; morbidity $\uparrow$ labour \\
  Discount rate & {\color{ogc}o$\to$c} & --- & economy's return $\to$ planning discount rate & shifts the build mix (via the loop) \\
  Energy demand & {\color{ogc}o$\to$c} & --- & economy's activity $\to$ energy demand & more capacity (via the loop) \\
  \bottomrule
  \end{tabularx}
  \par\smallskip
  {\color{sub}\scriptsize four \CLEWS$\to$\OGCore{} · two policy levers applied once
   to both sides · two \OGCore$\to$\CLEWS{}}
\end{frame}

\begin{frame}{Energy price --- the composite channel}
  \footnotesize The channel applies the \CLEWS{} electricity price as the composite of
  the previous slide: a cost-push on firms that use electricity as an input (the output
  sign) \emph{plus} a recycled consumption wedge on households (the incidence), with
  electricity's self-use netted out.
  \medskip
  \begin{itemize}\small
    \item households face $(1+\tauc_{i_e})\,p_{i_e}$, so demand responds --- the response
          \CLEWS{} lacks; and with energy a partial necessity the burden is
          \emph{regressive}: poorer households bear more even when the revenue is returned
          (recycled variant: $C_{ss}=-0.087\%$)
    \item the price is the scenario's \emph{real} electricity price, \textbf{near-flat}
          for the Philippines --- so the channel's effect is small ($Y_{ss}=-0.0005\%$,
          household side alone): transmitted honestly, the data does little
    \item the $\pm20\%$ and single-mechanism runs only stress-test the \emph{transmission}
  \end{itemize}
\end{frame}

\begin{frame}{Carbon price --- one price, two sides, counted once}
  \centering
  \begin{tikzpicture}[node distance=7mm]
    \node[pol] (p) {carbon price\\\$50/tCO$_2$};
    \node[sig, above right=1mm and 9mm of p] (c) {\CLEWS{}: emissions-penalty path\\$\Rightarrow$ the build mix decarbonises};
    \node[wdg, below right=1mm and 9mm of p] (o) {\OGCore{}: $\Delta\tauc_{i_e}=$ price $\times$ deflator\\$\times$ carbon intensity $/\,p^{b}_{i_e}$};
    \draw[ch] (p) -- (c); \draw[ch] (p) -- (o);
  \end{tikzpicture}
  \smallskip
  \begin{itemize}\footnotesize
    \item one chosen price enters \emph{both} models along the same path ---
          never read back a second time from the energy model's own implied
          carbon cost
    \item \OGCore{} firms buy no energy, so only \emph{household} energy carbon is
          priced on the macro side; industrial carbon is priced in \CLEWS{}
    \item revenue is recyclable (transfers); standalone it is the clearest
          contraction: $Y_{ss}=-0.040\%$
  \end{itemize}
  \smallskip
  {\raggedright\color{sub}\footnotesize the \emph{level} rides on the uncalibrated
   unit/deflator bridge --- mechanics validated, magnitude illustrative.\par}
\end{frame}

\begin{frame}{Public investment --- the grid on the government's books}
  \centering
  \chainrow{grid \& transmission\\build cost (\% of GDP)}{public-investment
    share $\alphaI$\\$\to$ public capital $\Kg$}{$\Kg$ raises all industries'
    output\\crowding-out if debt-financed}
  \medskip
  \begin{itemize}\small
    \item this is where crowding-out \emph{legitimately} lives: debt-financed
          spending competes for the savings pool, raises $r$, displaces private
          investment
    \item public-grid capital and private-generation capital touch different
          objects --- coupling both is complementary, not double counting
    \item in the Philippine scenario the \emph{extra} grid spending is $\approx 0$
          (the transition is generation-side), so the channel is inert here
          ($Y_{ss}=+0.000\%$); the mechanics scale with any non-zero spending
  \end{itemize}
\end{frame}

\begin{frame}{The capital pair I: capital intensity --- a worked example}
  \footnotesize
  \textbf{Signal:} reform/baseline ratio of the power fleet's capital-cost share
  $\;\Rightarrow\;$ scales the energy industry's capital exponent $\gammam$
  (here $+12.2\%$).\par\smallskip
  \textbf{The model's own first-order condition} (Cobb--Douglas, $\varepsilon=1$):
  the user cost $\rho_m$ and taxes cancel in the reform/baseline ratio, leaving an
  exact identity ---
  \[\textstyle
    K_m=\dfrac{\gammam\,p_m Y_m}{\rho_m}
    \qquad\Rightarrow\qquad
    \hat K_E=\hat\gamma_E\cdot\hat p_E\cdot\hat Y_E
  \]
  \centering
  \renewcommand{\arraystretch}{1.15}
  \begin{tabular}{@{}lrrrrr@{}}
  \toprule
   & $\hat\gamma_E$ & $\hat p_E$ & $\hat Y_E$ & $\hat K_E$ (obs.) & identity \\
  \midrule
  energy industry ($m_e=2$) & $+12.2\%$ & $-42.8\%$ & $+20.9\%$ & $-22.4\%$ & $-22.4\%$ \\
  \bottomrule
  \end{tabular}
  \par\smallskip
  \begin{itemize}
    \item the higher exponent cuts the sector's unit cost, and competition passes
          that into price $\to$ electricity \textbf{collapses} $-42.8\%$
    \item cheaper power lifts output $+20.9\%$ --- but revenue $p_mY_m$ still shrinks,
          so capital \textbf{falls} $-22.4\%$ (the identity closes to $0.03\%$)
  \end{itemize}
  \smallskip
  {\raggedright\color{claret}Raising the capital \emph{share} sheds capital: this is
  a factor-share and energy-price lever, \emph{not} a capital draw-in.\par}
\end{frame}

\begin{frame}{The capital pair II: the investment incentive --- the actual draw-in lever}
  \small
  The capital-\emph{attraction} force is a structurally different instrument: an
  \textbf{investment tax credit} on the energy industry.
  \[
    \rho_m=\frac{r+\delta-\taub_m\delta^{\tau}_m-
      {\color{policy}\tau^{\mathrm{inv}}_m}\delta}{1-\taub_m}
  \]
  \centering{\footnotesize the ITC enters the \emph{user-cost denominator}, where
  $\gammam$ is absent $\;\Rightarrow\;$ capital demand shifts \emph{out} at given $r$}
  \par\smallskip
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\small\bfseries the capital share $\gammam$ (capital intensity)}\par
      {\footnotesize moves factor \emph{demand} through the production exponent;
       absent from $\rho_m$. Result: cheaper electricity, \emph{less} energy capital.}
    \end{column}
    \begin{column}{0.48\textwidth}
      {\small\bfseries the tax credit (investment incentive)}\par
      {\footnotesize moves the \emph{cost of capital}; absent from the exponents.
       Result: capital reallocates \emph{into} energy. (A defined lever; not part of
       the Philippine reference run.)}
    \end{column}
  \end{columns}
  \smallskip
  \begin{alertblock}{The lesson}
    \footnotesize Two levers, two different objects --- \emph{not} two views of one
    build-out. Intuition says ``capital-intensive $\Rightarrow$ pulls capital in'';
    the model's own first-order condition shows the opposite. Don't stack them.
  \end{alertblock}
\end{frame}

\begin{frame}{Health --- cleaner air, fewer deaths, more productive workers}
  \centering
  \begin{tikzpicture}[node distance=5mm, every node/.append style={font=\scriptsize}]
    \node[sig] (e) {cleaner power:\\PM$_{2.5}$ emissions $-7.7\%$};
    \node[eff, right=of e] (m) {only $\sim$8\% reaches\\deaths ($M\approx0.082$)};
    \node[eff, right=of m] (d) {$\approx 279$ fewer\\deaths a year};
    \node[wdg, above right=-1mm and 7mm of d] (mo) {mortality: survival\\rates $\rho(s,t)$};
    \node[wdg, below right=-1mm and 7mm of d] (mb) {morbidity: worker\\productivity $e(t,s,j)$};
    \draw[ch] (e) -- (m); \draw[ch] (m) -- (d);
    \draw[ch] (d) -- (mo); \draw[ch] (d) -- (mb);
  \end{tikzpicture}
  \begin{itemize}\footnotesize
    \item real inputs: the Global Burden of Disease counts \textbf{43{,}951} deaths a year
          in the Philippines from breathing fine particulates (PM$_{2.5}$), with real
          age profiles
    \item cutting power-sector PM$_{2.5}$ by $7.7\%$ is \emph{not} $7.7\%$ fewer deaths:
          energy is one source among many, and deaths fall less than one-for-one as the
          air clears --- so only $\sim$\textbf{8\%} of the cut reaches mortality
          ($M\approx0.082$). That is \textbf{$\approx$279 fewer deaths}, not the
          $\approx$3{,}400 from blaming energy for \emph{every} pollution death
    \item \textbf{mortality}: pollution deaths fall mostly among the elderly, so the lives
          saved are mostly retirees --- long-run output barely moves
    \item \textbf{morbidity}: fewer working-age illnesses means more productive workers ---
          and \emph{that} is the real macro gain ($+0.19\%$ near-term; steady state
          $-0.003\%$)
  \end{itemize}
  \smallskip
  {\raggedright\color{claret}\footnotesize The signature --- barely negative in the long
  run, clearly positive along the way --- is a population responding to cleaner air, which
  a single-snapshot model cannot show. Still assumed: that deaths scale in a straight line
  with emissions, and the exact size of the productivity gain.\par}
\end{frame}

\begin{frame}{What the economy sends back: its return, its demand}
  Two channels run \emph{after} the economy solves and write ready-to-use
  \CLEWS{} inputs:
  \medskip
  \begin{itemize}\small\setlength\itemsep{6pt}
    \item \textbf{discount rate}: the economy's equilibrium return $\rp$ becomes the
          energy system's planning discount rate --- so energy planning uses the
          \emph{model-consistent} cost of capital. A lower return favours
          capital-heavy, long-lived assets.
    \item \textbf{energy demand}: how activity changed (industry output or
          consumption, with an elasticity) scales the demand the system must meet.
          In a single pass the change is $\approx$ zero --- latent by construction.
  \end{itemize}
  \medskip
  {\color{sub}\small They move the macro economy only once the loop iterates --- the
  quantity link of the classic coupled energy--macro models
  (\textsc{message-macro}, \textsc{times-macro}), but feeding an economy with
  households and a government instead of a single representative agent.}
\end{frame}

\dgmframe{Closing the loop}{loop}{%
  Re-solve the two models in turn until energy demand stops changing. Validated today:
  the single pass; naive price feedback can oscillate, so the loop needs damping.}

\section{Counting each signal once}

\begin{frame}{Five principles, each enforced by a guard}
  \small
  \begin{enumerate}\setlength\itemsep{6pt}
    \item \textbf{One carbon price.} It enters once --- a \CLEWS{} penalty and an
          \OGCore{} wedge on the same path --- never \emph{also} read back from the
          energy model's implied carbon cost.
    \item \textbf{One view of an energy-cost shock.} A price wedge, a carbon wedge,
          and a productivity haircut are alternative doors for the \emph{same}
          cost --- never stacked.
    \item \textbf{One view of generation capital.} A cost-push \emph{or} an
          investment credit --- and the capital share ($\gammam$) is \emph{not} a
          third interchangeable view: it is a factor-income lever that sheds
          capital (the worked example).
    \item \textbf{Revenue must be routed.} A $\tauc$ wedge mechanically raises
          consumption-tax revenue; recycle it (transfers, public investment) or
          finance it explicitly --- else a hidden fiscal expansion rides along.
    \item \textbf{Public and private capital are distinct.} $\alphaI\to\Kg$ and the
          firm's own $K$ touch different objects; coupling both is complementary.
  \end{enumerate}
\end{frame}
