\section{The coupling problem}

\begin{frame}{Two models, complementary gaps}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\color{clews}\bfseries CLEWS / OSeMOSYS}\par\smallskip
      {\small least-cost energy-system optimiser: technologies, dispatch, investment,
       and emissions}\par\smallskip
      {\small\begin{itemize}
        \item \textbf{rich:} technology, season-by-season operation, emissions, land \& water
        \item \textbf{gap:} energy demand is external --- no price response, no households,
              no government budget, no distributional accounts
      \end{itemize}}
    \end{column}
    \begin{column}{0.48\textwidth}
      {\color{ogc}\bfseries OG-Core}\par\smallskip
      {\small overlapping-generations general equilibrium: households, firms, and a
       government budget}\par\smallskip
      {\small\begin{itemize}
        \item \textbf{rich:} behaviour, saving, fiscal policy, incidence, demography
        \item \textbf{gap:} no energy technology --- firms do not use energy as a
              priced production input
      \end{itemize}}
    \end{column}
  \end{columns}
  \medskip
  \centering{\color{sub}\small The useful hand-off differs by direction:\\
  \textbf{quantities} flow from the economy, \textbf{prices and physical outcomes}
  flow from the energy system.}
\end{frame}

\dgmframe{The structural hand-off}{architecture}{%
  Quantities flow right (energy demand, the interest rate); prices and physical
  outcomes flow left (the energy price, build cost, emissions). The models stay
  separate and pass a few auditable signals.}

\begin{frame}{Why a \emph{staged soft} link, not a simultaneous hard link}
  \textbf{A simultaneous hard link is structurally awkward:}
  {\small\begin{itemize}
    \item \OGCore{} moves through the transition period by period; the energy model
          optimises the full planning horizon at once
    \item neither model fits inside the other: \OGCore{} has no single objective the
          energy model could be nested in, and no shock can be injected mid-solve
  \end{itemize}}
  \medskip
  \textbf{So the models stay separate and the link runs in stages:}
  \smallskip
  \begin{center}\small
  \resizebox{0.98\linewidth}{!}{%
  \begin{tikzpicture}[node distance=4mm]
    \node[wdg] (a) {solve \OGCore{}\\baseline};
    \node[sig, right=of a] (b) {solve \CLEWS{},\\extract signals};
    \node[wdg, right=of b] (c) {apply them to the\\economy; solve reform};
    \node[sig, right=of c] (d) {send results\\back to \CLEWS{}};
    \node[eff, right=of d] (e) {iterate to a\\fixed point};
    \draw[ch] (a) -- (b); \draw[ch] (b) -- (c); \draw[ch] (c) -- (d);
    \draw[ch, dashed] (d) -- (e);
  \end{tikzpicture}}
  \end{center}
  \smallskip
  {\color{sub}\small Current implementation: the single pass (solid). The loop (dashed)
  is designed, and the return signals already exist, but the outer iteration is not yet
  implemented.}
\end{frame}

\section{Signals, wedges, and the main price signal}

\begin{frame}{The vocabulary: signal $\to$ wedge $\to$ channel}
  \centering
  \renewcommand{\arraystretch}{1.35}
  \begin{tabularx}{0.92\textwidth}{@{}>{\bfseries}l >{\raggedright\arraybackslash}X@{}}
  {\color{clews}Signal} & an outcome read from a \emph{solved} energy scenario:
    electricity price, grid investment, or emissions \\
  {\color{ogc}Wedge} & the \OGCore{} parameter that carries it: $\tauc$, $Z_m$,
    $\alphaI$, $\rho(s,t)$, or $e(t,s,j)$ \\
  Channel & one signal mapped to one wedge, with a diagnostic check for large or
    poorly defined mappings \\
  \end{tabularx}
  \medskip

  {\color{sub}\small The country-specific pieces, such as the electricity industry, are
  read from the country's own model at run time.}
\end{frame}

\begin{frame}{The main hand-off is the electricity price}
  \footnotesize The most important \CLEWS{} signal is the \textbf{electricity price}: the
  levelized cost of a delivered unit --- capital, running costs, and fuel, averaged over
  the solved system's generation. It is the right hand-off because it lets the
  economy do two things the energy model cannot do on its own:
  \smallskip
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\small\bfseries 1. It lets demand respond}\par
      {\footnotesize In \CLEWS{}, demand is an external input. In \OGCore{}, households
      change energy use when prices change, subject to a subsistence floor.}
    \end{column}
    \begin{column}{0.48\textwidth}
      {\small\bfseries 2. It shows who pays}\par
      {\footnotesize A single ``system cost'' is silent on who bears it. \OGCore{}'s
      households, by age and income, split the burden --- and because energy is a
      partial necessity, a price rise falls hardest on the poor.}
    \end{column}
  \end{columns}
  \smallskip
  {\raggedright\color{sub}\footnotesize In later iterations, the models move toward
  agreement between the electricity price and the cost of supplying the demand generated
  by the economy.\par}
\end{frame}

\begin{frame}{The energy channel: two uses, one transmission}
  \small
  \OGCore{} firms use capital and labour, not energy. The electricity-price signal is
  therefore carried through the two places electricity appears in the accounts.
  \medskip

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\color{clews}\bfseries Firms}\par\smallskip
      {\footnotesize Electricity as an intermediate input, about three quarters of use in
      the Philippine accounts. The channel applies a cost-push to industry production
      costs, which sets the output sign.}
    \end{column}
    \begin{column}{0.48\textwidth}
      {\color{ogc}\bfseries Households}\par\smallskip
      {\footnotesize Electricity as final consumption. The channel applies a
      consumption-tax wedge on the energy good; when revenue recycling is on, the
      revenue is rebated as transfers.}
    \end{column}
  \end{columns}
  \medskip

  {\footnotesize The electricity industry's own electricity purchases are netted out to
  avoid double counting. The input side remains reduced-form until energy is added as a
  priced production input.}
\end{frame}

\section{The eight channels}

\begin{frame}{The eight channels at a glance}
  \centering\scriptsize
  \renewcommand{\arraystretch}{1.02}
  \begin{tabularx}{\textwidth}{@{}l c >{\raggedright\arraybackslash}X >{\raggedright\arraybackslash}X >{\raggedright\arraybackslash}X@{}}
  \toprule
  \textbf{Channel} & \textbf{Dir.} & \textbf{Signal / lever} & \textbf{\OGCore{} object} & \textbf{First-order effect} \\
  \midrule
  Energy price & {\color{clews}C$\to$O} & electricity price (levelized cost / index) & cost-push on firms + consumption wedge & output $\downarrow$; regressive incidence \\
  Carbon price & {\color{policy}policy} & carbon price (\$/tCO$_2$) & energy-wedge add-on + emissions penalty & demand $\downarrow$; emissions $\downarrow$ \\
  Public investment & {\color{clews}C$\to$O} & grid \& transmission build cost & public-investment share $\to$ public capital & public capital $\uparrow$; crowding-out, debt \\
  Capital intensity & {\color{clews}C$\to$O} & power-fleet capital-cost share & energy industry's capital share ($\gammam$) & energy price $\downarrow$; energy $K$ $\downarrow$ \\
  Investment incentive & {\color{policy}policy} & tax-credit rate & cost of capital (investment tax credit) & capital reallocates into energy \\
  Health & {\color{clews}C$\to$O} & PM$_{2.5}$ emissions change & survival rates + worker productivity & mortality $\approx 0$; labour productivity $\uparrow$ \\
  Discount rate & {\color{ogc}O$\to$C} & economy's return & planning discount rate & shifts the build mix via iteration \\
  Energy demand & {\color{ogc}O$\to$C} & economy's activity & specified annual demand & changes capacity via iteration \\
  \bottomrule
  \end{tabularx}
  \par\smallskip
  {\color{sub}\scriptsize four \CLEWS$\to$\OGCore{} · two policy levers applied once
   to both sides · two \OGCore$\to$\CLEWS{}}
\end{frame}

\begin{frame}{Energy price --- one price, two economic effects}
  \small
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\bfseries Firm cost-push}\par\smallskip
      {\footnotesize\begin{itemize}\setlength\itemsep{4pt}
        \item electricity-intensive industries get a productivity haircut
        \item the haircut is scaled by each industry's electricity input share
        \item this is the output and wage side of the price shock
      \end{itemize}}
    \end{column}
    \begin{column}{0.46\textwidth}
      {\bfseries Household incidence}\par\smallskip
      {\footnotesize\begin{itemize}\setlength\itemsep{4pt}
        \item direct household energy consumption gets a consumption wedge
        \item the wedge is diluted by electricity's share of the energy good
        \item revenue is rebated as transfers when revenue recycling is on
      \end{itemize}}
    \end{column}
  \end{columns}
  \medskip

  {\color{sub}\footnotesize The two pieces are one composite channel in the coupled run.
  A separate cost-push number is a diagnostic decomposition; the mechanism itself applies
  both pieces before solving the economy.}
\end{frame}

\begin{frame}{Carbon price --- one policy, counted once}
  \centering
  \begin{tikzpicture}[node distance=7mm]
    \node[pol] (p) {carbon price\\\$50/tCO$_2$};
    \node[sig, above right=1mm and 9mm of p] (c) {\CLEWS{}: emissions-penalty path\\$\Rightarrow$ the build mix decarbonises};
    \node[wdg, below right=1mm and 9mm of p] (o) {\OGCore{}: $\Delta\tauc_{i_e}=$ price $\times$ unit conversion\\$\times$ carbon intensity $/\,p^{b}_{i_e}$};
    \draw[ch] (p) -- (c); \draw[ch] (p) -- (o);
  \end{tikzpicture}
  \smallskip
  \begin{itemize}\footnotesize
    \item one chosen price enters \emph{both} models along the same policy path; it is
          not also inferred from the energy model's own implied carbon cost
    \item \OGCore{} firms buy no energy, so only \emph{household} energy carbon is
          priced on the macro side; industrial carbon is priced in \CLEWS{}
    \item revenue can be recycled as transfers; applied by itself, this channel changes
          steady-state output by $-0.021\%$
  \end{itemize}
  \smallskip
  {\raggedright\color{sub}\footnotesize that level is illustrative because the conversion
   from \CLEWS{} monetary units to the \OGCore{} numéraire is not yet calibrated.\par}
\end{frame}

\begin{frame}{Public investment --- grid costs enter through public capital}
  \centering
  \chainrow{grid \& transmission\\build cost (\% of GDP)}{public-investment
    share $\alphaI$\\$\to$ public capital $\Kg$}{$\Kg$ raises all industries'
    output\\crowding-out if debt-financed}
  \medskip
  \begin{itemize}\small
    \item this is where crowding-out enters the linked model: debt-financed
          spending competes for the savings pool, raises $r$, displaces private
          investment
    \item public-grid capital and private-generation capital are different objects;
          using both is complementary, not double counting
    \item in the Philippine scenario the \emph{extra} grid spending is $\approx 0$
          because the transition is mostly generation-side, so this channel has
          essentially no effect here; the steady-state output change rounds to
          $+0.000\%$. The mechanics scale with any non-zero spending
  \end{itemize}
\end{frame}

\begin{frame}{Capital intensity --- why the sign is not automatic}
  \small
  \textbf{Signal:} \CLEWS{} says the clean power fleet is more capital-heavy. In
  \OGCore{}, this is represented as a higher energy-sector capital share, not as a new
  investment subsidy or spending program.\par\medskip

  \centering
  \chainrow{capital share rises\\factor mix changes}{unit cost and\\electricity price fall}{sector revenue can fall\\even if output rises}
  \medskip

  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\bfseries What to watch}\par\smallskip
      {\footnotesize Energy capital follows the sector's revenue and user cost. If the
      price drop is larger than the output gain, the sector can use less capital even
      though the technology is more capital-intensive.}
    \end{column}
    \begin{column}{0.46\textwidth}
      {\bfseries Outputs to update}\par\smallskip
      {\footnotesize\begin{itemize}\setlength\itemsep{3pt}
        \item energy capital share
        \item energy price
        \item energy output
        \item energy capital stock
      \end{itemize}}
    \end{column}
  \end{columns}
  \medskip

  {\raggedright\color{claret}\footnotesize This is a factor-mix and price channel. The
  capital-pull lever is the investment incentive, which lowers the cost of capital.\par}
\end{frame}

\begin{frame}{Investment incentive --- the actual capital-pull lever}
  \small
  The lever that directly attracts capital is an \textbf{investment tax credit} for the
  energy industry.
  \[
    \rho_m=\frac{r+\delta-\taub_m\delta^{\tau}_m-
      {\color{policy}\tau^{\mathrm{inv}}_m}\delta}{1-\taub_m}
  \]
  \centering{\footnotesize The credit lowers $\rho_m$, where $\gammam$ is absent; capital
  demand shifts \emph{out} at a given interest rate.}
  \par\medskip
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\small\bfseries Capital intensity}\par
      {\footnotesize changes the factor mix through $\gammam$; the solved effect on
      energy capital depends on energy price, output, and user cost.}
    \end{column}
    \begin{column}{0.48\textwidth}
      {\small\bfseries Investment incentive}\par
      {\footnotesize changes the cost of capital through $\rho_m$; capital reallocates
      \emph{into} energy. This lever is defined but not in the Philippine reference run.}
    \end{column}
  \end{columns}
  \medskip
  \begin{alertblock}{The lesson}
    \footnotesize Two levers, two different objects. Capital intensity changes the factor
    mix; the tax credit changes the cost of capital. They can move energy capital in
    opposite directions.
  \end{alertblock}
\end{frame}

\begin{frame}{Health --- mortality is small, productivity matters}
  \centering
  \begin{tikzpicture}[node distance=5mm, every node/.append style={font=\scriptsize}]
    \node[sig] (e) {cleaner power:\\PM$_{2.5}$ emissions $-2.7\%$};
    \node[eff, right=of e] (m) {only $\sim$8\% reaches\\deaths ($M\approx0.082$)};
    \node[eff, right=of m] (d) {$\approx 95$ fewer\\deaths a year};
    \node[wdg, above right=-1mm and 7mm of d] (mo) {mortality: survival\\rates $\rho(s,t)$};
    \node[wdg, below right=-1mm and 7mm of d] (mb) {morbidity: worker\\productivity $e(t,s,j)$};
    \draw[ch] (e) -- (m); \draw[ch] (m) -- (d);
    \draw[ch] (d) -- (mo); \draw[ch] (d) -- (mb);
  \end{tikzpicture}
  \medskip
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      {\small\bfseries Scale adjustment}\par
      {\footnotesize $43{,}951 \times 0.082 \times (-0.0265)\approx -95$ deaths.
      The adjustment keeps the emissions cut from being read as all pollution deaths.}
    \end{column}
    \begin{column}{0.48\textwidth}
      {\small\bfseries Macro effect}\par
      {\footnotesize Mortality changes are retiree-heavy, so steady-state output barely
      moves ($-0.0003\%$). Working-age morbidity gives the near-term gain ($+0.19\%$).}
    \end{column}
  \end{columns}
  \medskip
  {\raggedright\color{claret}\footnotesize Time profile: productivity lifts the
  transition; retiree-heavy survival effects dominate the long run.\par}
\end{frame}

\begin{frame}{Return and demand go back to CLEWS}
  Two channels run \emph{after} the economy solves and write ready-to-use
  \CLEWS{} inputs:
  \medskip
  \begin{itemize}\small\setlength\itemsep{6pt}
    \item \textbf{discount rate}: the economy's equilibrium return $\rp$ becomes the
          energy system's planning discount rate --- so energy planning uses the
          \emph{model-consistent} cost of capital. A lower return favours
          technologies with high up-front costs and long operating lives.
    \item \textbf{energy demand}: how activity changed (industry output or
          consumption, with an elasticity) scales the demand the system must meet.
          In a single pass the change is $\approx$ zero, so this channel mainly matters
          once the models iterate.
  \end{itemize}
  \medskip
  {\color{sub}\small They affect the macro economy only once the loop iterates: the
  quantity link of the classic coupled energy--macro models
  (\textsc{message-macro}, \textsc{times-macro}), but feeding an economy with
  households and a government instead of a single representative agent.}
\end{frame}

\dgmframe{Closing the loop}{loop}{%
  Iterate until energy demand stops changing. Current implementation: the single pass;
  the loop still needs damping.}
